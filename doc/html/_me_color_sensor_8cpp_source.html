642
643 r = Redvalue>>8;
644 g = Greenvalue>>8;

646
647 if((r == g) && (g == b))
648 {
649 h = 0;
650 }
651 else
652 {
653 max = MAX(r,g,b);
654 min = MIN(r,g,b);
655 c = max - min;
656 if(max == r)
657 {
658 h = ((g-b) / c % 6)*60;
659 }
660 else if(max == g)
661 {
662 h = ((b-r) / c + 2) * 60;
663 }
664 else if(min == b)
665 {
666 h = ((r-g) / c + 4)*60;
667 }
668 }

879int8_t MeColorSensor::readData(uint8_t start, uint8_t *buffer, uint8_t size)
880{
881 int16_t i = 0;
882 int8_t return_value = 0;
883 Wire.beginTransmission(Device_Address);
884 return_value = Wire.write(start);
885
886 if(return_value != 1)
887 {
888 return(I2C_ERROR);
889 }
890 return_value = Wire.endTransmission(false);
891
892 if(return_value != 0)
893 {
894 return(return_value);
895 }
896 delayMicroseconds(1);
897 /* Third parameter is true: relase I2C-bus after data is read. */
898 Wire.requestFrom(Device_Address, size, (uint8_t)true);
899
900 while(Wire.available() && i < size)
901 {
902 buffer[i++] = Wire.read();
903 }
904 delayMicroseconds(1);
905
906 if(i != size)
907 {
908 return(I2C_ERROR);
909 }
910 return 0;
911}
