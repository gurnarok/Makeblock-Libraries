222uint8_t MeOneWire::reset(void)
223{
224 MeIO_REG_TYPE mask = bitmask;
225 volatile MeIO_REG_TYPE *reg MeIO_REG_ASM = baseReg;
226 uint8_t r;
227 uint8_t retries = 125;
228
229 noInterrupts();
230 MeDIRECT_MODE_INPUT(reg, mask);
231 interrupts();
232 /* wait until the wire is high... just in case */
233 do
234 {
235 if (--retries == 0)
236 {
237 return(0);
238 }
239 delayMicroseconds(2);
240 }
241 while (!MeDIRECT_READ(reg, mask));
242
243 noInterrupts();
244 MeDIRECT_WRITE_LOW(reg, mask);
245 MeDIRECT_MODE_OUTPUT(reg, mask); /* drive output low */
246 interrupts();
247 delayMicroseconds(480);
248 noInterrupts();
249 MeDIRECT_MODE_INPUT(reg, mask); /* allow it to float */
250 delayMicroseconds(70);
251 r = !MeDIRECT_READ(reg, mask);
252 interrupts();
253 delayMicroseconds(410);
254 return(r);
255}

592uint8_t MeOneWire::search(uint8_t *newAddr)
593{
594 uint8_t id_bit_number;
595 uint8_t last_zero, rom_byte_number, search_result;
596 uint8_t id_bit, cmp_id_bit;
597
598 uint8_t rom_byte_mask, search_direction;
599
600 /* initialize for search */
601 id_bit_number = 1;
602 last_zero = 0;
603 rom_byte_number = 0;
604 rom_byte_mask = 1;
605 search_result = 0;
606
607 /* if the last call was not the last one */
608 if (!LastDeviceFlag)
609 {
610 /* 1-Wire reset */
611 if (!reset())
612 {
613 /* reset the search */
614 LastDiscrepancy = 0;
615 LastDeviceFlag = false;
616 LastFamilyDiscrepancy = 0;
617 return(false);
618 }
619
620 /* issue the search command */
621 write(0xF0);
622
623 /* loop to do the search */
624 do
625 {
626 /* read a bit and its complement */
627 id_bit = read_bit();
628 cmp_id_bit = read_bit();
629
630 /* check for no devices on 1-wire */
631 if ((id_bit == 1) && (cmp_id_bit == 1))
632 {
633 break;
634 }
635 else
636 {
637 /* all devices coupled have 0 or 1 */
638 if (id_bit != cmp_id_bit)
639 {
640 search_direction = id_bit; /* bit write value for search */
641 }
642 else
643 {
644 /*
645 if this discrepancy if before the Last Discrepancy
646 on a previous next then pick the same as last time
647 */
648 if (id_bit_number < LastDiscrepancy)
649 {
650 search_direction = ((ROM_NO[rom_byte_number] & rom_byte_mask) > 0);
651 }
652 else
653 {
654 /* if equal to last pick 1, if not then pick 0 */
655 search_direction = (id_bit_number == LastDiscrepancy);
656 }
657
658 /* if 0 was picked then record its position in LastZero */
659 if (search_direction == 0)
660 {
661 last_zero = id_bit_number;
662
663 /* check for Last discrepancy in family */
664 if (last_zero < 9)
665 {
666 LastFamilyDiscrepancy = last_zero;
667 }
668 }
669 }
670
671 /*
672 set or clear the bit in the ROM byte rom_byte_number
673 with mask rom_byte_mask
674 */
675 if (search_direction == 1)
676 {
677 ROM_NO[rom_byte_number] |= rom_byte_mask;
678 }
679 else
680 {
681 ROM_NO[rom_byte_number] &= ~rom_byte_mask;
682 }
683
684 /* serial number search direction write bit */
685 write_bit(search_direction);
686
687 /*
688 increment the byte counter id_bit_number
689 and shift the mask rom_byte_mask
690 */
691 id_bit_number++;
692 rom_byte_mask <<= 1;
693
694 /* if the mask is 0 then go to new SerialNum byte rom_byte_number and reset mask */
695 if (rom_byte_mask == 0)
696 {
697 rom_byte_number++;
698 rom_byte_mask = 1;
699 }
700 }
701 }
702 while (rom_byte_number < 8); /* loop until through all ROM bytes 0-7 */
703
704 /* if the search was successful then */
705 if (!(id_bit_number < 65))
706 {
707 /* search successful so set LastDiscrepancy,LastDeviceFlag,search_result */
708 LastDiscrepancy = last_zero;
709
710 /* check for last device */
711 if (LastDiscrepancy == 0)
712 {
713 LastDeviceFlag = true;
714 }
715
716 search_result = true;
717 }
718 }
719
720 /* if no device found then reset counters so next 'search' will be like a first */
721 if (!search_result || !ROM_NO[0])
722 {
723 LastDiscrepancy = 0;
724 LastDeviceFlag = false;
725 LastFamilyDiscrepancy = 0;
726 search_result = false;
727 }
728 for (int16_t i = 0; i < 8; i++)
729 {
730 newAddr[i] = ROM_NO[i];
731 }
732 return(search_result);
733}
