334boolean MeStepperOnBoard::runSpeed(void)
335{
336 // Dont do anything unless we actually have a step interval
337 if (!_stepInterval)
338 {
339 return false;
340 }
341
342 if (micros() - _lastStepTime > _stepInterval)
343 {
344 if (_dir == DIRECTION_CW)
345 {
346 // Clockwise
347 _currentPos += 1;
348 }
349 else
350 {
351 // Anticlockwise
352 _currentPos -= 1;
353 }
354 step();
355 _lastStepTime = micros();
356 return true;
357 }
358 else
359 {
360 return false;
361 }

456void MeStepperOnBoard::computeNewSpeed(void)
457{
458 long distanceTo = distanceToGo();
459 long stepsToStop = (long)((_speed * _speed) / (2.0 * _acceleration));
460 if (distanceTo == 0 && stepsToStop <= 1)
461 {
462 // We are at the target and its time to stop
463 _stepInterval = 0;
464 _speed = 0.0;
465 _n = 0;
466 return;
467 }
468
469 if (distanceTo > 0)
470 {
471 // We are anticlockwise from the target
472 // Need to go clockwise from here, maybe decelerate now
473 if (_n > 0)
474 {
475 // Currently accelerating, need to decel now? Or maybe going the wrong way?
476 if ((stepsToStop >= distanceTo) || _dir == DIRECTION_CCW)
477 {
478 _n = -stepsToStop; // Start deceleration
479 }
480 }
481 else if (_n < 0)
482 {
483 // Currently decelerating, need to accel again?
484 if ((stepsToStop < distanceTo) && _dir == DIRECTION_CW)
485 {
486 _n = -_n; // Start accceleration
487 }
488 }
489 }
490 else if (distanceTo < 0)
491 {
492 // We are clockwise from the target
493 // Need to go anticlockwise from here, maybe decelerate
494 if (_n > 0)
495 {
496 // Currently accelerating, need to decel now? Or maybe going the wrong way?
497 if ((stepsToStop >= -distanceTo) || _dir == DIRECTION_CW)
498 {
499 _n = -stepsToStop; // Start deceleration
500 }
501 }
502 else if (_n < 0)
503 {
504 // Currently decelerating, need to accel again?
505 if ((stepsToStop < -distanceTo) && _dir == DIRECTION_CCW)
506 {
507 _n = -_n; // Start accceleration
508 }
509 }
510 }
511
512 // Need to accelerate or decelerate
513 if (_n == 0)
514 {
515 // First step from stopped
516 _cn = _c0;
517 _dir = (distanceTo > 0) ? DIRECTION_CW : DIRECTION_CCW;
518 }
519 else
520 {
521 // Subsequent step. Works for accel (n is +_ve) and decel (n is -ve).
522 _cn = _cn - ((2.0 * _cn) / ((4.0 * _n) + 1)); // Equation 13
523 _cn = max(_cn, _cmin);
524 }
525 _n++;
526 _stepInterval = _cn;
527 _speed = 1000000.0 / _cn;
528 if (_dir == DIRECTION_CCW)
529 {
530 _speed = -_speed;
531 }
532}
