552void Me7SegmentDisplay::display(double value, uint8_t digits)
553{
554Posotion_1:
555 uint8_t buf[4] = { ' ', ' ', ' ', ' ' };
556 uint8_t tempBuf[4];
557 uint8_t b = 0;
558 uint8_t bit_num = 0;
559 uint8_t int_num = 0;
560 uint8_t isNeg = 0;
561 double number = value;
562 if (number >= 9999.5 || number <= -999.5)
563 {
564 buf[0] = ' ';
565 buf[1] = ' ';
566 buf[2] = ' ';
567 buf[3] = 0x0e;
568 }
569 else
570 {
571 // Handle negative numbers
572 if (number < 0.0)
573 {
574 number = -number;
575 isNeg = 1;
576 }
577 // Round correctly so that print(1.999, 2) prints as "2.00"
578 double rounding = 0.5;
579 for (uint8_t i = 0; i < digits; ++i)
580 {
581 rounding /= 10.0;
582 }
583 number += rounding;
584
585 // Extract the integer part of the number and print it
586 uint16_t int_part = (uint16_t)number;
587 double remainder = number - (double)int_part;
588 do
589 {
590 uint16_t m = int_part;
591 int_part /= 10;
592 int8_t c = m - 10 * int_part;
593 tempBuf[int_num] = c;
594 int_num++;
595 }
596 while (int_part);
597
598 bit_num = isNeg + int_num + digits;
599
600 if (bit_num > 4)
601 {
602 bit_num = 4;
603 digits = 4 - (isNeg + int_num);
604 goto Posotion_1;
605 }
606 b = 4 - bit_num;
607 if (isNeg)
608 {
609 buf[b++] = 0x21; // '-' display minus sign
610 }
611 for (uint8_t i = int_num; i > 0; i--)
612 {
613 buf[b++] = tempBuf[i - 1];
614 }
615 // Print the decimal point, but only if there are digits beyond
616 if (digits > 0)
617 {
618 buf[b - 1] += 0x10; // display '.'
619 // Extract digits from the remainder one at a time
620 while (digits-- > 0)
621 {
622 remainder *= 10.0;
623 int16_t toPrint = int16_t(remainder);
624 buf[b++] = toPrint;
625 remainder -= toPrint;
626 }
627 }
628 }
629 display(buf);
630}
