441int16_t MeSerial::poll(void)
442{
443 int16_t val = 0;
444 int16_t bitDelay = _bitPeriod - clockCyclesToMicroseconds(50);
445 if (digitalRead(_RxPin) == LOW)
446 {
447 for (int16_t offset = 0; offset < 8; offset++)
448 {
449 delayMicroseconds(bitDelay);
450 val |= digitalRead(_RxPin) << offset;
451 }
452 delayMicroseconds(bitDelay);
453 return (val & 0xff);
454 }
455 return (-1);
456}
