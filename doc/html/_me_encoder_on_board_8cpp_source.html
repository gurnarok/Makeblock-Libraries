807int16_t MeEncoderOnBoard::pidPositionToPwm(void)
808{
809 float seek_speed = 0;
810 float pos_error = 0;
811 float speed_error = 0;
812 float d_component = 0;
813 float out_put_offset = 0;
814
815 pos_error = distanceToGo();
816
817 if((_Lock_flag == false) && (_Dir_lock_flag == true) && (pos_error < 0))
818 {
819 d_component = encode_structure.currentSpeed;
820 out_put_offset = encode_structure.PID_pos.D * d_component;
821 encode_structure.PID_pos.Output = -out_put_offset;
822 _Encoder_output = encode_structure.PID_pos.Output;
823 _Lock_flag = true;
824 encode_structure.currentPwm = _Encoder_output;
825 return _Encoder_output;
826 }
827 else if((_Lock_flag == false) && (_Dir_lock_flag == false) && (pos_error > 0))
828 {
829 d_component = encode_structure.currentSpeed;
830 out_put_offset = encode_structure.PID_pos.D * d_component;
831 encode_structure.PID_pos.Output = -out_put_offset;
832 _Encoder_output = encode_structure.PID_pos.Output;
833 _Lock_flag = true;
834 encode_structure.currentPwm = _Encoder_output;
835 return _Encoder_output;
836 }
837
838 //speed pid;
839 if((_Lock_flag == false) && (abs(pos_error) >= encode_structure.targetSpeed * DECELERATION_DISTANCE_PITCH))
840 {
841 speed_error = encode_structure.currentSpeed - encode_structure.targetSpeed * (pos_error/abs(pos_error));
842 out_put_offset = encode_structure.PID_speed.P * speed_error;
843 out_put_offset = constrain(out_put_offset,-25,25);
844 encode_structure.PID_speed.Output = _Encoder_output;
845 encode_structure.PID_speed.Output -= out_put_offset;
846 encode_structure.PID_speed.Output = constrain(encode_structure.PID_speed.Output,-255,255);
847 _Encoder_output = encode_structure.PID_speed.Output;
848 }
849 //position pid;
850 else
851 {
852 if((_Lock_flag == false) && (abs(pos_error) > ENCODER_POS_DEADBAND))
853 {
854 seek_speed = sqrt(abs(encode_structure.targetSpeed * DECELERATION_DISTANCE_PITCH * (pos_error-ENCODER_POS_DEADBAND)))/DECELERATION_DISTANCE_PITCH;
855 d_component = encode_structure.currentSpeed - seek_speed * (pos_error/abs(pos_error));
856 out_put_offset = encode_structure.PID_pos.D * d_component;
857 out_put_offset = constrain(out_put_offset,-20,20);
858
859 encode_structure.PID_pos.Output = _Encoder_output;
860 encode_structure.PID_pos.Output -= out_put_offset;
861 if(pos_error >= 0)
862 {
863 encode_structure.PID_pos.Output = constrain(encode_structure.PID_pos.Output,PWM_MIN_OFFSET,255);
864 }
865 else
866 {
867 encode_structure.PID_pos.Output = constrain(encode_structure.PID_pos.Output,-255,-PWM_MIN_OFFSET);
868 }
869 _Encoder_output = encode_structure.PID_pos.Output;
870 }
871 else
872 {
873 _Lock_flag = true;
874 if((_callback != NULL) && (_Callback_flag == false))
875 {
876 _Callback_flag = true;
877 _callback(_Slot,_extId);
878 }
879 d_component = encode_structure.currentSpeed;
880 out_put_offset = encode_structure.PID_pos.D * d_component;
881 out_put_offset = constrain(out_put_offset,-20,20);
882 encode_structure.PID_pos.Output = pos_error * encode_structure.PID_pos.P;
883 encode_structure.PID_pos.Output -= out_put_offset;
884 encode_structure.PID_pos.Output = constrain(encode_structure.PID_pos.Output,-255,255);
885 _Encoder_output = encode_structure.PID_pos.Output;
886 }
887 }
