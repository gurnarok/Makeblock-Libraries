70ISR(TIMER_INTR_NAME)
71{
72 uint8_t irdata = (uint8_t)digitalRead(2);
73 // uint32_t new_time = micros();
74 // uint8_t timer = (new_time - irparams.lastTime)>>6;
75 irparams.timer++; // One more 50us tick
76 if (irparams.rawlen >= RAWBUF)
77 {
78 // Buffer overflow
79 irparams.rcvstate = STATE_STOP;
80 }
81 switch(irparams.rcvstate)
82 {
83 case STATE_IDLE: // In the middle of a gap
84 if (irdata == MARK)
85 {
86 irparams.rawlen = 0;
87 irparams.timer = 0;
88 irparams.rcvstate = STATE_MARK;
89 }
90 break;
91 case STATE_MARK: // timing MARK
92 if (irdata == SPACE)
93 {
94 // MARK ended, record time
95 irparams.rawbuf[irparams.rawlen++] = irparams.timer;
96 irparams.timer = 0;
97 irparams.rcvstate = STATE_SPACE;
98 }
99 break;
100 case STATE_SPACE: // timing SPACE
101 if (irdata == MARK)
102 {
103 // SPACE just ended, record it
104 irparams.rawbuf[irparams.rawlen++] = irparams.timer;
105 irparams.timer = 0;
106 irparams.rcvstate = STATE_MARK;
107 }
108 else
109 { // SPACE
110 if (irparams.timer > GAP_TICKS)
111 {
112 // big SPACE, indicates gap between codes
113 // Mark current code as ready for processing
114 // Switch to STOP
115 // Don't reset timer; keep counting space width
116 irparams.rcvstate = STATE_STOP;
117 irparams.lastTime = millis();
118 }
119 }
120 break;
121 case STATE_STOP: // waiting, measuring gap
122 if(millis() - irparams.lastTime > 200)
123 {
124 irparams.rawlen = 0;
125 irparams.timer = 0;
126 irparams.rcvstate = STATE_IDLE;
127 }
128 else if (irdata == MARK)
129 {
130 // reset gap timer
131 irparams.timer = 0;
132 }
133 break;
134 }

265// NECs have a repeat only 4 items long
266ErrorStatus MeIR::decodeNEC()
267{
268 uint32_t data = 0;
269 int offset = 0; // Skip first space
270 // Initial mark
271 if (!MATCH(rawbuf[offset], NEC_HDR_MARK/50))
272 {
273 return ERROR;
274 }
275 offset++;
276 // Check for repeat
277 if (rawlen == 3 &&
278 MATCH(rawbuf[offset], NEC_RPT_SPACE/50) &&
279 MATCH(rawbuf[offset+1], NEC_BIT_MARK/50))
280 {
281 bits = 0;
282 // results->value = REPEAT;
283 // Serial.println("REPEAT");
284 decode_type = NEC;
285 return SUCCESS;
286 }
287 if (rawlen < (2 * NEC_BITS + 3))
288 {
289 return ERROR;
290 }
291 // Initial space
292 if (!MATCH(rawbuf[offset], NEC_HDR_SPACE/50))
293 {
294 return ERROR;
295 }
296 offset++;
297 for (int i = 0; i < NEC_BITS; i++)
298 {
299 if (!MATCH(rawbuf[offset], NEC_BIT_MARK/50))
300 {
301 return ERROR;
302 }
303 offset++;
304 if (MATCH(rawbuf[offset], NEC_ONE_SPACE/50))
305 {
306 //data = (data << 1) | 1;
307 data = (data >> 1) | 0x80000000;
308 }
309 else if (MATCH(rawbuf[offset], NEC_ZERO_SPACE/50))
310 {
311 //data <<= 1;
312 data >>= 1;
313 }
314 else
315 {
316 return ERROR;
317 }
318 offset++;
319 }
320 // Success
321 bits = NEC_BITS;
322 value = data;
323 decode_type = NEC;
324 return SUCCESS;
325}
