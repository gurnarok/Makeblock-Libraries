93uint32_t MeHost_Pack(uint8_t * buf,
94 uint32_t bufSize,
95 uint8_t module,
96 uint8_t * data,
97 uint32_t length)
98{
99 uint32_t i = 0;
100
101 // head: 0xA5
102 buf[i++] = 0xA5;
103 buf[i++] = module;
104 // pack length
105 buf[i++] = *((uint8_t *)&length + 0);
106 buf[i++] = *((uint8_t *)&length + 1);
107 buf[i++] = *((uint8_t *)&length + 2);
108 buf[i++] = *((uint8_t *)&length + 3);
109 // pack data
110 for(uint32_t j = 0; j < length; ++j)
111 {
112 buf[i++] = data[j];
113 }
114
115 // calculate the LRC
116 uint8_t check = 0x00;
117 for(uint32_t j = 0; j < length; ++j)
118 {
119 check ^= data[j];
120 }
121 buf[i++] = check;
122
123 // tail: 0x5A
124 buf[i++] = 0x5A;
125
126 if (i > bufSize)
127 {
128 return 0;
129 }
130 else
131 {
132 return i;
133 }
134}
