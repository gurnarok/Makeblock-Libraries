693int16_t MeUSBHost::probeDevice()
694{
695 int16_t c;
696 if(!ch375_online){
697 CH375_WR( CMD_CHECK_EXIST );
698 CH375_WR( 0x5A);
699 c = CH375_RD(); // should return 0xA5
700 if(c!=0xA5) return 0;
701 ch375_online = true;
702 resetBus();
703 }
704
705 c = getIrq();
706 if(c!=USB_INT_CONNECT) return 0;
707 resetBus(); // reset bus and wait the device online again
708 c=0;
709 while(c!=USB_INT_CONNECT){
710 delay(500); // some device may need long time to get ready
711 c = getIrq();
712#ifdef CH375_DBG
713 Serial.print("waiting:");
714 Serial.println(c,HEX);
715#endif
716 }
717 if( initHIDDevice()==1)
718 device_online=true;
719}
