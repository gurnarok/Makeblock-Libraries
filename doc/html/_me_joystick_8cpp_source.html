234float MeJoystick::angle(void)
235{
236 int16_t x_value;
237 int16_t y_value;
238 float angle;
239#ifdef ME_PORT_DEFINED
240 x_value = MePort::aRead1();
241 y_value = MePort::aRead2();
242#else // ME_PORT_DEFINED
243 x_value = analogRead(_X_port);
244 y_value = analogRead(_Y_port);
245#endif // ME_PORT_DEFINED
246 x_value = (x_value - CENTER_VALUE) + _X_offset;
247 y_value = (y_value - CENTER_VALUE) + _Y_offset;
248 if((abs(x_value) < 10) && (abs(y_value) < 10))
249 {
250 angle = 0;
251 }
252 else
253 {
254 angle = atan2(x_value, y_value) * 180.0 / PI;
255 }
256 return angle;
257}
