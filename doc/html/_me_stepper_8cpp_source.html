308void MeStepper::computeNewSpeed()
309{
310 long distanceTo = distanceToGo();
311 long stepsToStop = (long)((_speed * _speed) / (2.0 * _acceleration));
312 if (distanceTo == 0 && stepsToStop <= 1)
313 {
314 // We are at the target and its time to stop
315 _stepInterval = 0;
316 _speed = 0.0;
317 _n = 0;
318 return;
319 }
320
321 if (distanceTo > 0)
322 {
323 // We are anticlockwise from the target
324 // Need to go clockwise from here, maybe decelerate now
325 if (_n > 0)
326 {
327 // Currently accelerating, need to decel now? Or maybe going the wrong way?
328 if ((stepsToStop >= distanceTo) || _dir == DIRECTION_CCW)
329 {
330 _n = -stepsToStop; // Start deceleration
331 }
332 }
333 else if (_n < 0)
334 {
335 // Currently decelerating, need to accel again?
336 if ((stepsToStop < distanceTo) && _dir == DIRECTION_CW)
337 {
338 _n = -_n; // Start accceleration
339 }
340 }
341 }
342 else if (distanceTo < 0)
343 {
344 // We are clockwise from the target
345 // Need to go anticlockwise from here, maybe decelerate
346 if (_n > 0)
347 {
348 // Currently accelerating, need to decel now? Or maybe going the wrong way?
349 if ((stepsToStop >= -distanceTo) || _dir == DIRECTION_CW)
350 {
351 _n = -stepsToStop; // Start deceleration
352 }
353 }
354 else if (_n < 0)
355 {
356 // Currently decelerating, need to accel again?
357 if ((stepsToStop < -distanceTo) && _dir == DIRECTION_CCW)
358 {
359 _n = -_n; // Start accceleration
360 }
361 }
362 }
363
364 // Need to accelerate or decelerate
365 if (_n == 0)
366 {
367 // First step from stopped
368 _cn = _c0;
369 _dir = (distanceTo > 0) ? DIRECTION_CW : DIRECTION_CCW;
370 }
371 else
372 {
373 // Subsequent step. Works for accel (n is +_ve) and decel (n is -ve).
374 _cn = _cn - ((2.0 * _cn) / ((4.0 * _n) + 1)); // Equation 13
375 _cn = max(_cn, _cmin);
376 }
377 _n++;
378 _stepInterval = _cn;
379 _speed = 1000000.0 / _cn;
380 if (_dir == DIRECTION_CCW)
381 {
382 _speed = -_speed;
383 }
384}
