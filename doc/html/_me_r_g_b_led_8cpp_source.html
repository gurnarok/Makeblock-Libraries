531/*
532 This routine writes an array of bytes with RGB values to the Dataout pin
533 using the fast 800kHz clockless WS2811/2812 protocol.
534 */
535/* Timing in ns */
536#define w_zeropulse (350)
537#define w_onepulse (900)
538#define w_totalperiod (1250)
539
540/* Fixed cycles used by the inner loop */
541#define w_fixedlow (3)
542#define w_fixedhigh (6)
543#define w_fixedtotal (10)
544
545/* Insert NOPs to match the timing, if possible */
546#define w_zerocycles ( ( (F_CPU / 1000) * w_zeropulse) / 1000000)
547#define w_onecycles ( ( (F_CPU / 1000) * w_onepulse + 500000) / 1000000)
548#define w_totalcycles ( ( (F_CPU / 1000) * w_totalperiod + 500000) / 1000000)
549
550/* w1 - nops between rising edge and falling edge - low */
551#define w1 (w_zerocycles - w_fixedlow)
552/* w2 nops between fe low and fe high */
553#define w2 (w_onecycles - w_fixedhigh - w1)
554/* w3 nops to complete loop */
555#define w3 (w_totalcycles - w_fixedtotal - w1 - w2)
556
557#if w1 > 0
558#define w1_nops w1
559#else
560#define w1_nops 0
561#endif
562
563/*
564 The only critical timing parameter is the minimum pulse length of the "0"
565 Warn or throw error if this timing can not be met with current F_CPU settings.
566 */
567#define w_lowtime ( (w1_nops + w_fixedlow) * 1000000) / (F_CPU / 1000)
568#if w_lowtime > 550
569#error "Light_ws2812: Sorry, the clock speed is too low. Did you set F_CPU correctly?"
570#elif w_lowtime > 450
571#warning "Light_ws2812: The timing is critical and may only work on WS2812B, not on WS2812(S)."
572#warning "Please consider a higher clockspeed, if possible"
573#endif

614void MeRGBLed::rgbled_sendarray_mask(uint8_t *data, uint16_t datlen, uint8_t maskhi, uint8_t *port)
615{
616 uint8_t curbyte, ctr, masklo;
617 uint8_t oldSREG = SREG;
618 cli(); // Disables all interrupts
619
620 masklo = *port & ~maskhi;
621 maskhi = *port | maskhi;
622
623 while(datlen--)
624 {
625 curbyte = *data++;
626
627 asm volatile (
628 " ldi %0,8 \n\t"
629 "loop%=: \n\t"
630 " st X,%3 \n\t" // '1' [02] '0' [02] - re
631#if (w1_nops & 1)
632 w_nop1
633#endif
634#if (w1_nops & 2)
635 w_nop2
636#endif
637#if (w1_nops & 4)
638 w_nop4
639#endif
640#if (w1_nops & 8)
641 w_nop8
642#endif
643#if (w1_nops & 16)
644 w_nop16
645#endif
646 " sbrs %1,7 \n\t" // '1' [04] '0' [03]
647 " st X,%4 \n\t" // '1' [--] '0' [05] - fe-low
648 " lsl %1 \n\t" // '1' [05] '0' [06]
649#if (w2_nops & 1)
650 w_nop1
651#endif
652#if (w2_nops & 2)
653 w_nop2
654#endif
655#if (w2_nops & 4)
656 w_nop4
657#endif
658#if (w2_nops & 8)
659 w_nop8
660#endif
661#if (w2_nops & 16)
662 w_nop16
663#endif
664 " brcc skipone%= \n\t" /* '1' [+1] '0' [+2] - */
665 " st X,%4 \n\t" /* '1' [+3] '0' [--] - fe-high */
666 "skipone%=: " /* '1' [+3] '0' [+2] - */
667
668#if (w3_nops & 1)
669 w_nop1
670#endif
671#if (w3_nops & 2)
672 w_nop2
673#endif
674#if (w3_nops & 4)
675 w_nop4
676#endif
677#if (w3_nops & 8)
678 w_nop8
679#endif
680#if (w3_nops & 16)
681 w_nop16
682#endif
683
684 " dec %0 \n\t" // '1' [+4] '0' [+3]
685 " brne loop%=\n\t" // '1' [+5] '0' [+4]
686 : "=&d" (ctr)
687 : "r" (curbyte), "x" (port), "r" (maskhi), "r" (masklo)
688 );
689 }
690
691 SREG = oldSREG;
692}
