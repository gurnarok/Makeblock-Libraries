207void MeGyro::update(void)
208{
209 static unsigned long last_time = 0;
210 int8_t return_value;
211 double dt, filter_coefficient;
212 /* read imu data */
213 return_value = readData(0x3b, i2cData, 14);
214 if(return_value != 0)
215 {
216 return;
217 }
218
219 double ax, ay;
220 /* assemble 16 bit sensor data */
221 accX = ( (i2cData[0] << 8) | i2cData[1] );
222 accY = ( (i2cData[2] << 8) | i2cData[3] );
223 accZ = ( (i2cData[4] << 8) | i2cData[5] );
224 gyrX = ( ( (i2cData[8] << 8) | i2cData[9] ) - gyrXoffs) / gSensitivity;
225 gyrY = ( ( (i2cData[10] << 8) | i2cData[11] ) - gyrYoffs) / gSensitivity;
226 gyrZ = ( ( (i2cData[12] << 8) | i2cData[13] ) - gyrZoffs) / gSensitivity;
227 ax = atan2(accX, sqrt( pow(accY, 2) + pow(accZ, 2) ) ) * 180 / 3.1415926;
228 ay = atan2(accY, sqrt( pow(accX, 2) + pow(accZ, 2) ) ) * 180 / 3.1415926;
229
230 dt = (double)(millis() - last_time) / 1000;
231 last_time = millis();
232
233 if(accZ > 0)
234 {
235 gx = gx - gyrY * dt;
236 gy = gy + gyrX * dt;
237 }
238 else
239 {
240 gx = gx + gyrY * dt;
241 gy = gy - gyrX * dt;
242 }
243 gz += gyrZ * dt;
244 gz = gz - 360 * floor(gz / 360);
245 if(gz > 180)
246 {
247 gz = gz - 360;
248 }
249
250 /*
251 complementary filter
252 set 0.5sec = tau = dt * A / (1 - A)
253 so A = tau / (tau + dt)
254 */
255 filter_coefficient = 0.5 / (0.5 + dt);
256 gx = gx * filter_coefficient + ax * (1 - filter_coefficient);
257 gy = gy * filter_coefficient + ay * (1 - filter_coefficient);
258}

484void MeGyro::deviceCalibration(void)
485{
486 int8_t return_value;
487 uint16_t x = 0;
488 uint16_t num = 500;
489 long xSum = 0, ySum = 0, zSum = 0;
490 for(x = 0; x < num; x++)
491 {
492 return_value = readData(0x43, i2cData, 6);
493 xSum += ( (i2cData[0] << 8) | i2cData[1] );
494 ySum += ( (i2cData[2] << 8) | i2cData[3] );
495 zSum += ( (i2cData[4] << 8) | i2cData[5] );
496 }
497 gyrXoffs = xSum / num;
498 gyrYoffs = ySum / num;
499 gyrZoffs = zSum / num;
500}

557int8_t MeGyro::readData(uint8_t start, uint8_t *buffer, uint8_t size)
558{
559 int16_t i = 0;
560 int8_t return_value = 0;
561 Wire.beginTransmission(Device_Address);
562 return_value = Wire.write(start);
563 if(return_value != 1)
564 {
565 return(I2C_ERROR);
566 }
567 return_value = Wire.endTransmission(false);
568 if(return_value != 0)
569 {
570 return(return_value);
571 }
572 delayMicroseconds(1);
573 /* Third parameter is true: relase I2C-bus after data is read. */
574 Wire.requestFrom(Device_Address, size, (uint8_t)true);
575 while(Wire.available() && i < size)
576 {
577 buffer[i++] = Wire.read();
578 }
579 delayMicroseconds(1);
580 if(i != size)
581 {
582 return(I2C_ERROR);
583 }
584 return(0); //return: no error
585}
