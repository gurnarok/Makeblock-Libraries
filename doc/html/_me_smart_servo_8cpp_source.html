1117void MeSmartServo::smartServoEventHandle(void)
1118{
1119 while (available())
1120 {
1121 // get the new byte:
1122 uint8_t inputData = read();
1123 if(parsingSysex)
1124 {
1125 if (inputData == END_SYSEX)
1126 {
1127 //stop sysex byte
1128 parsingSysex = false;
1129 //fire off handler function
1130 processSysexMessage();
1131 }
1132 else
1133 {
1134 //normal data byte - add to buffer
1135 sysex.storedInputData[sysexBytesRead] = inputData;
1136 sysexBytesRead++;
1137 if(sysexBytesRead > DEFAULT_UART_BUF_SIZE-1)
1138 {
1139 parsingSysex = false;
1140 sysexBytesRead = 0;
1141 }
1142 }
1143 }
1144 else if(inputData == START_SYSEX)
1145 {
1146 parsingSysex = true;

1194void MeSmartServo::smartServoCmdResponse(void *arg)
1195{
1196 long angle_v;
1197 float speed_v;
1198 float temp_v;
1199 float vol_v;
1200 float current_v;
1201 uint8_t servoNum = sysex.val.dev_id;
1202 int16_t cmd = (int16_t)sysex.val.value[0];
1203 switch(cmd)
1204 {
1205 case GET_SERVO_CUR_ANGLE:
1206 angle_v = readLong(sysex.val.value,1);
1207 servo_dev_list[sysex.val.dev_id - 1].angleValue = angle_v;
1208 // Serial.print("servo angle: ");
1209 // Serial.println(angle_v);
1210 resFlag |= 0x02;
1211 break;
1212 case GET_SERVO_SPEED:
1213 speed_v = readFloat(sysex.val.value,1);
